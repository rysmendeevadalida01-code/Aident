from flask import Flask, request, jsonify, render_template_string import re app = Flask(__name__) # --- МИНИМАЛИСТИЧНЫЙ ИНТЕРФЕЙС (APPLE STYLE) --- HTML_TEMPLATE = ''' AI Detector

AI DETECTOR

0% Робот
0% Человек
''' # --- УЛУЧШЕННЫЙ АЛГОРИТМ АНАЛИЗА --- @app.route('/') def index(): return render_template_string(HTML_TEMPLATE) @app.route('/analyze', methods=['POST']) def analyze(): text = request.json.get('text', '') words = re.findall(r'\w+', text.lower()) sentences = re.split(r'[.!?]+', text) sentences = [s.strip() for s in sentences if len(s.strip()) > 2] if not words or not sentences: return jsonify({'ai': 0, 'human': 100}) # 1. Анализ уникальности слов (TTR - Type-Token Ratio) # ИИ часто использует ограниченный набор "идеальных" слов. unique_ratio = len(set(words)) / len(words) # 2. Анализ вариативности длины предложений (Burstiness) # Человек пишет: "Коротко. Потом очень длинно и витиевато. Опять коротко." # ИИ пишет: "Среднее предложение. Еще одно среднее. И еще одно." lens = [len(s.split()) for s in sentences] avg_len = sum(lens) / len(lens) variance = sum((x - avg_len) ** 2 for x in lens) / len(lens) if len(lens) > 1 else 0 # Расчет вероятности (упрощенная модель детекции) ai_score = 50 # Если вариативность низкая (скучный ритм) -> это ИИ if variance < 15: ai_score += 25 else: ai_score -= 15 # Если словарный запас бедный -> это ИИ if unique_ratio < 0.5: ai_score += 20 else: ai_score -= 10 # Финальные границы ai_score = min(max(int(ai_score), 2), 98) return jsonify({ 'ai': ai_score, 'human': 100 - ai_score }) if __name__ == '__main__': app.run(debug=True)
