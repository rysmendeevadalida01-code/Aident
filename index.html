from flask import Flask, request, jsonify, render_template_string app = Flask(__name__) # --- МИНИМАЛИСТИЧНЫЙ ДИЗАЙН --- HTML_TEMPLATE = ''' AI Detector

AI DETECTOR

Робот 0%
Человек 0%
''' # --- ЛОГИКА (УЛЬТРА-БЫСТРАЯ) --- @app.route('/') def home(): return render_template_string(HTML_TEMPLATE) @app.route('/analyze', methods=['POST']) def analyze(): text = request.json.get('text', '') # Алгоритм мгновенного анализа вариативности words = text.split() if not words: return jsonify({'ai': 0, 'human': 0}) unique_words = len(set(words)) / len(words) avg_word_len = sum(len(w) for w in words) / len(words) # ИИ обычно использует меньше уникальных слов и среднюю длину 5.2 if 4.5 < avg_word_len < 6.0 and unique_words < 0.6: ai_score = 82 else: ai_score = 14 return jsonify({'ai': ai_score, 'human': 100 - ai_score}) if __name__ == '__main__': app.run(debug=True)
