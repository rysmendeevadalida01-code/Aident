from flask import Flask, request, jsonify, render_template_string import re app = Flask(__name__) # --- МАКСИМАЛЬНО ЧИСТЫЙ ДИЗАЙН БЕЗ ЛИШНЕГО ТЕКСТА --- HTML_TEMPLATE = ''' AI Detector

AI DETECTOR

0% Робот
0% Человек
''' # --- ЛОГИКА АНАЛИЗА (БЫСТРАЯ И БЕЗ ОШИБОК) --- @app.route('/') def index(): return render_template_string(HTML_TEMPLATE) @app.route('/analyze', methods=['POST']) def analyze(): data = request.json text = data.get('text', '') # 1. Считаем вариативность (Burstiness) sentences = re.split(r'[.!?]+', text) sentences = [s.strip() for s in sentences if len(s) > 2] if len(sentences) < 1: return jsonify({'ai': 0, 'human': 100}) lens = [len(s.split()) for s in sentences] avg_len = sum(lens) / len(lens) # Если все предложения одной длины — это ИИ variance = sum((x - avg_len) ** 2 for x in lens) / len(lens) if len(lens) > 1 else 0 # 2. Логика определения ai_prob = 50 if variance < 10: ai_prob += 35 # Очень ровный текст if len(set(text.split())) / len(text.split()) < 0.5: ai_prob += 10 # Мало уникальных слов ai_final = min(max(int(ai_prob), 1), 99) return jsonify({ 'ai': ai_final, 'human': 100 - ai_final }) if __name__ == '__main__': # Порт 5000 может быть занят системой, если не работает — поменяй на 5001 app.run(debug=True, port=5000)
